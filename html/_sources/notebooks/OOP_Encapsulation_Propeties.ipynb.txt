    "## @properties solves the problem"
   ]
  },
  {
   "cell_type": "code",
   "execution_count": 42,
   "metadata": {},
   "outputs": [],
   "source": [
    "class P:\n",
    "    def __init__(self, x):\n",
    "        self.x = x # self.x is the @x.setter\n",
    "    \n",
    "    @property\n",
    "    def x(self):\n",
    "        return self.__x # this is a private variable\n",
    "    \n",
    "    @x.setter\n",
    "    def x(self, x):     \n",
    "        if x > 1000:\n",
    "            self.__x = 1000\n",
    "        elif x < 0:\n",
    "            self.__x = 0\n",
    "        else:\n",
    "            self.__x = x"
